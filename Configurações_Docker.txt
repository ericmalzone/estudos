-- baixar a imagem da Cloudera
docker pull cloudera/quickstart:latest

-- subindo um container com a imagem da Cloudera QuickStart
sudo docker run --hostname=quickstart.cloudera --privileged=true -t -i -p 7180:7180 -p 8888:8888 -p 2222:22 docker.io/cloudera/quickstart:latest /usr/bin/docker-quickstart

-- após subir a imagem, ver o serviço da cloudera
service cloudera-scm-server status

-- abrir um shell a parte e dar telnet na porta
telnet localhost 8888

-- acima estou utilizando a porta 2222 do meu notebook para redirecionar o acesso para a porta 22 do container

-- instalar ssh em um container docker
yum install openssh-server

-- inicializar o ssh
/etc/init.d/sshd start


-------- Criando uma imagem Docker em cima de um container --------

-- Primeiro verifique qual seria o id do container (ativo)
docker ps  

-- Depois utilize o comando de commit
docker commit <id_container>

-- Após isso você pode verificar que ele já estará nas imagens docker de sua máquina
-- Mas com as colunas (REPOSITORY e TAG) vazias, com o valor '<none>'
docker images

-- Neste caso apenas falta você dar um nome para esta TAG, passando como parâmetro o
-- 'IMAGE ID' para que sua nova imagem docker em cima do container fique completa
docker tag <IMAGE ID> <nome_desejado>
